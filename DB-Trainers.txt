type Training {
    _id: String!
    type: String!
    coach: String!
    user: String!
    date: Date
    maxDate: Date
    description: String
    trainingBlocks: [String]
    registryDate: Date
    completed: Boolean!
  }
type TrainingBlock {
    _id: String!
    coach: String!
    description: String
    distance: Int
    duration: Int
    maxHR: Int
    minHR: Int
    maxSpeed: Float
    minSpeed: Float
    altitude: Float
    result: [TrainingBlockResult]
    schema: Boolean
  }
  type TrainingBlockResult {
    date: Date!
    distance: Float!
    coords: LocationPoint!
    altitude: Float!
    HR: Float!
    course: Float!
    speed: Float!
    gyro: XYZCoords!
    accel: XYZCoords!
    magn: XYZCoords!
  }
